START

CALL MESAJE1;

BE MATRIX A(10,10);
BE MATRIX B(10,10);

BE INT dimensionAx;
dimensionAx = 1; 
BE INT dimensionAy;
dimensionAy = 2;
BE INT dimensionBx;
dimensionBx = 3;
BE INT dimensionBy;
dimensionBy = 4;


WHILE((dimensionAx!=dimensionBx)&(dimensionAy!=dimensionBy))
    WRITE "ingrese Ax"; READ dimensionAx;
    WRITE "ingrese Ay"; READ dimensionAy;
    WRITE "ingrese Bx"; READ dimensionBx;
    WRITE "ingrese By"; READ dimensionBy;
    
ENDWHILE

WRITE "ingrese la matriz A";

BE INT i; BE INT leeA;
BE INT j;
WHILE(i<dimensionAx)

    j = 0;

    WHILE(j<dimensionAy)
        READ  leeA;
        A(i,j) = leeA;
        j = j+1;
    ENDWHILE

ENDWHILE

i = 0; 
BE INT leeB;

FOR((i<dimensionBx),i=i+1;)
j = 0; 
    FOR((j<dimensionBy),j=j+1;)
        READ leeB;
        B(i,j) = leeB;
    ENDFOR
ENDFOR

i = 0; 
FOR((i<dimensionBx), i = i+1;)
j = 0;
    FOR((j<dimensionBy), j = j+1;)
        WRITE A(i,j) + B(i,j);
    ENDFOR
ENDFOR
CALL MENSAJE2;

FINISH

ROUTINE(MENSAJE1)
    WRITE "INICIO";
RETURN

ROUTINE(MENSAJE2)
    WRITE "FIN";
RETURN
